```js
const tree = {
  data: 1,
  left: {
    data: 2,
    left: null,
    right: null,
  },
  right: {
    data: 3,
    left: null,
    right: null,
  },
}

function traverseTree(node) {
  if (!node) return;
  traverseTree(node.left);
  console.log(node.data);
  traverseTree(node.right);
}

function bitTree() {
  traverseTree(tree);
}

function bitTreeNoTraverse(tree) {
  const stack = [];
  let temp = tree;
  while (temp || stack.length > 0) {
    if (temp) {
      stack.push(temp);
      temp = temp.left;
    } else {
      temp = stack.pop();
      console.log(temp.data);
      temp = temp.right;
    }

  }
}

bitTreeNoTraverse(tree)
```
